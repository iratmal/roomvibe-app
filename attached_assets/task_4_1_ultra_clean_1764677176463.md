Scope
1. Create data source for premium rooms
- Add a file, e.g. data/premiumRooms.ts.
- Export an array of ~100 room objects with this shape and generator:

```ts
export type PremiumRoom = {
  id: string;
  name: string;
  category: string;
  imageUrl: string;
};

const categories = ["living_room", "bedroom", "office", "gallery"];

export const premiumRooms: PremiumRoom[] = Array.from({ length: 100 }, (_, i) => {
  const index = i + 1;
  return {
    id: `PRM_ROOM_${index.toString().padStart(2, "0")}`,
    name: `Premium Room ${index}`,
    category: categories[i % categories.length],
    imageUrl: "/images/rooms/premium/default.jpg",
  };
});
```
